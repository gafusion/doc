\section{Transport problem}

Steady state, for each energy channel $\sigma \in \{e,i\}$:
%
\begin{equation}
\frac{1}{\vpr(r)} \frac{\partial}{\partial r}
 \left[ \vpr(r) \, Q_\sigma(r) \right] = S_\sigma(r) \; .
\label{eq.transport}
\end{equation}
%
Integrating gives the target flux
%
\begin{equation}
Q^T_\sigma(r) = \frac{P_\sigma(r)}{\vpr(r)} \; ,
\qquad
P_\sigma(r) = \int_0^r dx \, \vpr(x) \, S_\sigma(x) \; ,
\label{eq.target}
\end{equation}
%
so only the integrated powers enter.  The sources depend on the
profiles, $S_\sigma = S_\sigma(T)$, through fusion heating,
electron-ion exchange and radiation.  The transport flux is
%
\begin{equation}
Q_\sigma = Q_\sigma^{\rm turb} + Q_\sigma^{\rm neo} \; ,
\end{equation}
%
with $Q^{\rm turb}_\sigma$ from TGLF and, at present,
$Q^{\rm neo}_\sigma = 0$.

The channels couple only pointwise, through the local flux and the
sources.  We therefore drop $\sigma$: $z$, $T$, $Q$ and $A$ are read as
vectors over the $n_c$ channels at each radius, and every operation
below acts on them componentwise.

\section{Normalization}

\begin{equation}
\qgb = n_e T_e c_s \left( \frac{\rho_s}{a} \right)^{2} \; ,
\qquad
c_s = \sqrt{\frac{T_e}{m_D}} \; ,
\qquad
\rho_s = \frac{c_s}{\Omega_{ci}} = \frac{m_D c_s c}{e B_{\rm unit}} \; ,
\end{equation}
%
with $a$ the minor radius of the last closed flux surface and $\Omega_{ci}
= e B_{\rm unit}/(m_D c)$ the deuterium cyclotron frequency (Gaussian
units).  Explicitly,
%
\begin{equation}
\qgb = \frac{m_D^{1/2} c^2}{e^2 a^2 B_{\rm unit}^2} \; n_e \, T_e^{5/2} \; .
\label{eq.qgb}
\end{equation}
%
Writing $\qhat \doteq Q/\qgb$, the normalized target is
%
\begin{equation}
\qhat^T = \frac{P}{\vpr \, \qgb} \; ,
\label{eq.qhattarget}
\end{equation}
%
which depends on the solution through $\qgb$ even when $S$ does not.

\section{Scale-length formulation}

With $x \doteq r/a$, the unknown is
%
\begin{equation}
z(x) \doteq - \frac{\partial \ln T}{\partial x} \; .
\label{eq.zdef}
\end{equation}
%
Integrating (\ref{eq.zdef}) outward from the origin,
%
\begin{equation}
\ln T(x) = \ln A - \int_0^{x} dx^\prime \, z(x^\prime) \; ,
\qquad
A \doteq T(0) \; ,
\label{eq.integrate}
\end{equation}
%
so $z$ determines the profile up to the single constant $A$.

\subsection{Axis condition}

$T$ is even in $x$, so $z$ is odd:
%
\begin{equation}
z(0) = 0 \; .
\label{eq.axis}
\end{equation}
%
This is imposed, not solved for.

\subsection{Grid and reintegration}

\begin{equation}
x_0 = 0 \; , \qquad x_1 < x_2 < \dots < x_N \; ,
\label{eq.grid}
\end{equation}
%
non-uniform, with resolution concentrated toward the edge.  The
integral in (\ref{eq.integrate}) is a linear functional of $z$, so for
any quadrature with weights $W$,
%
\begin{equation}
\boxed{\;
T_j(z,A) = A \exp \left( - \sum_{m=0}^{j} W_{jm} \, z_m \right) \; ,
\qquad z_0 = 0 \; . }
\label{eq.W}
\end{equation}
%
$W_{jm}$ is the cumulative quadrature weight for $\int_0^{x_j}$, so
$W_{jm} = 0$ for $m > j$: $W$ is lower triangular, and $T_j$ depends on
$z$ only at radii interior to $x_j$.  Positivity of $T$ is automatic.

\section{Boundary condition}
\label{sec.bc}

Any {\it linear} condition on the discrete profile $T = \{ T_j \}$ may
be written, without loss of generality, as
%
\begin{equation}
B \cdot T = 1 \; ,
\label{eq.linbc}
\end{equation}
%
with $B$ a known vector.  By (\ref{eq.W}), $T$ is linear in $A$, so
(\ref{eq.linbc}) determines $A$:
%
\begin{equation}
\boxed{\;
A = \left[ \sum_{j} B_j \, \exp \left(
 - \sum_{m=0}^{j} W_{jm} \, z_m \right) \right]^{-1} \; }
\label{eq.profile}
\end{equation}

\subsection{The classic case}

Write $T_*$ for the prescribed boundary value at $x_N$.  The Dirichlet
condition $T_N = T_*$ is $B_j = \delta_{jN}/T_*$.  Only the $j=N$ term
survives in (\ref{eq.profile}),
%
\begin{equation}
A = T_* \exp \left( \sum_{m=0}^{N} W_{Nm} \, z_m \right) \; ,
\end{equation}
%
and substituting into (\ref{eq.W}),
%
\begin{equation}
T_j = T_* \exp \left( \sum_{m=0}^{N}
 \bigl( W_{Nm} - W_{jm} \bigr) \, z_m \right) \; .
\label{eq.dirichlet}
\end{equation}
%
For the trapezoidal rule the cumulative weights are
%
\begin{equation}
W_{j0} = \frac{x_1-x_0}{2} \; , \qquad
W_{jm} = \frac{x_{m+1}-x_{m-1}}{2} \quad (0 < m < j) \; , \qquad
W_{jj} = \frac{x_j-x_{j-1}}{2} \; ,
\label{eq.trap}
\end{equation}
%
with $W_{jm} = 0$ for $m > j$.  The difference $W_{Nm} - W_{jm}$
collapses to the panels outside $x_j$, and (\ref{eq.dirichlet}) is
%
\begin{equation}
\boxed{\;
T_j = T_* \exp \left( \sum_{m=j+1}^{N}
 \frac{x_m - x_{m-1}}{2} \bigl( z_m + z_{m-1} \bigr) \right) \; }
\label{eq.dirichlettrap}
\end{equation}
%
the trapezoidal quadrature of $\int_{x_j}^{x_N} dx \, z$.  $T_j$ is
independent of $z$ interior to $x_j$, and (\ref{eq.axis}) never enters.

\section{Discrete problem}

For one channel the unknown is
%
\begin{equation}
z = \{ z_j \} \in \mathbb{R}^{N} \; ,
\qquad
z_j \doteq z(x_j) \; ,
\end{equation}
%
and the $N$ equations to be solved are
%
\begin{equation}
\boxed{\;
\qhat_j(z) = \qhat^T_j(z) \; }
\label{eq.match}
\end{equation}
%
both sides evaluated on the profile (\ref{eq.profile}) built from the
same $z$.  Over all channels the system has dimension $N n_c$.

\section{Structure}
\label{sec.structure}

The flux is local in the pair $(z_j,T_j)$,
%
\begin{equation}
\qhat_j = \qhat \bigl( z_j ;\, T_j \bigr) \; ,
\label{eq.localflux}
\end{equation}
%
with density, geometry and field fixed.  $T_j$ enters through $\tau =
T_i/T_e$, $\beta_e \propto n_e T_e/B_{\rm unit}^2$, $\nu_e \propto n_e
T_e^{-2}$, the Debye length, and $\qgb \propto n_e T_e^{5/2}$: weak, but
nonzero.

The locality is in $(z_j,T_j)$, not in $z$.  By (\ref{eq.W}) and
(\ref{eq.profile}), $T_j$ depends on $z$ interior to $x_j$ through $W$,
and on all of $z$ through $A$.  Hence $\qhat_j$ is a nonlinear, nonlocal
functional of $z$, and (\ref{eq.match}) is coupled across the grid.

The target is explicitly nonlocal,
%
\begin{equation}
\qhat^T_j = \frac{1}{\vpr_j \, \qgbj}
 \int_0^{x_j} dx \, \vpr(x) \, S(x) \; ,
\label{eq.nonlocaltarget}
\end{equation}
%
a functional of $T$ at every radius interior to $x_j$, since $S = S(T)$.

\section{A learned inverse}
\label{sec.inverse}

Existing surrogate methods (PORTALS) fit the {\it forward} map
(\ref{eq.localflux}), $z \mapsto \qhat$, and solve
%
\begin{equation}
\qhat^{\rm model}_j (z) = \qhat^T_j
\end{equation}
%
for $z$ by root-finding at every iteration.  Consider fitting the {\it
inverse} map instead.  At fixed $T_j$, (\ref{eq.localflux}) is $n_c$
equations in the $n_c$ unknowns $z_j$, and may be inverted:
%
\begin{equation}
\boxed{\;
z_j = \zeta \bigl( \qhat_j,\, T_j \bigr) \; }
\label{eq.inverse}
\end{equation}
%
$\zeta$ returns the gradient that carries a given flux.  Since $\qhat^T$
is known, the matching gradient is obtained by {\it evaluating} $\zeta$,
and (\ref{eq.match}) becomes a fixed point in the profile alone,
%
\begin{equation}
z^{(k+1)}_j = \zeta \bigl( \qhat^T_j(z^{(k)}),\, T_j(z^{(k)}) \bigr)
 \; ,
\label{eq.picard}
\end{equation}
%
with $T(z)$ from (\ref{eq.W}) and (\ref{eq.profile}).  Were $T$ exact,
one pass would suffice: the iteration resolves only the nonlocality of
$T_j(z)$, which (\ref{eq.inverse}) holds fixed, and whose effect on
$\qhat$ is weak.

The training data are those of the forward map: each flux evaluation at
$x_j$ yields $(z_j,T_j) \to \qhat_j$, read as input $(\qhat,T)$ and
output $z$.  $\zeta$ is a fixed function of its $2 n_c$ arguments, and
may be trained offline.

The flux is stiff: small below a critical gradient, rising steeply above
it.  $\qhat(z)$ is steep, so an error in a fitted $\qhat$ is amplified
on inversion; $\zeta(\qhat)$ is flat, and an error in $\zeta$ is not.
But $\zeta$ is single-valued only above the critical gradient: below it
$\qhat \simeq 0$ and the inverse is ill-posed.  The contraction rate of
(\ref{eq.picard}) is unmeasured.

\section{Edge boundary condition}

EGYRO solves to the separatrix.  ELM-free and without a pedestal, there
is no pedestal top at which to pin $T$, and $T_*$ is no longer
prescribed: it is set by parallel transport in the scrape-off layer,
%
\begin{equation}
T_* = f \bigl( P_{\rm sep} \bigr) \; ,
\qquad
T_* \sim P_{\rm sep}^{2/7} \; ,
\label{eq.sol}
\end{equation}
%
the exponent following from $q_\parallel = -\kappa_0 T^{5/2} \partial_s
T$.  But the power crossing the separatrix,
%
\begin{equation}
P_{\rm sep} = \int_0^1 dx \, \vpr(x) \, S(x) \; ,
\end{equation}
%
is a functional of the profile, since $S = S(T)$.  Hence $B$ in
(\ref{eq.linbc}) depends on $z$, and (\ref{eq.profile}) is unavailable:
$A$ can no longer be eliminated.  Promote it to an unknown and append
(\ref{eq.sol}),
%
\begin{align}
\qhat_j(z,A) = &~ \qhat^T_j(z,A) \; ,
 && N n_c \ \text{equations} \; , \\
T_N(z,A) - f \bigl( P_{\rm sep}[T(z,A)] \bigr)
 = &~ 0 \; , && n_c \ \text{equations} \; ,
\label{eq.augmented}
\end{align}
%
with $(z,A) \in \mathbb{R}^{N n_c + n_c}$.  The appended rows require no
flux evaluation, and the $2/7$ exponent makes the added nonlinearity
weak.
